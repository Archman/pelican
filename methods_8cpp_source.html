122  au = 0.934*bfield*100*lambdau/sqrt(2); // normalized undulator parameter, calculated from the input parameters read from namelist file
123  ku = 2*PI/lambdau; // undulator wave number
124  omegas = 2*PI*C0/lambdas; // radiation angular frequency
125  gammar = sqrt(lambdau*(1+au*au)/2.0/lambdas); // resonant gamma
126  sigmax = sqrt(avgbeta*emitn/gamma0);
127  j0 = current/(PI*sigmax*sigmax); // transverse current density

132  totalIntSteps = (int) (num/deltz); // total integration steps
